[844. 比较含退格的字符串 - 力扣（LeetCode）](https://leetcode.cn/problems/backspace-string-compare/)



## 栈实现（用数组实现的栈）

```java
    class Solution {
        char[] ans1;
        char[] ans2;
        int size1;
        int size2;
        int maxLength = 210;

        public boolean backspaceCompare(String s, String t) {
            ans1 = new char[maxLength];
            ans2 = new char[maxLength];
            size1 = 0;
            size2 = 0;
            for (int i = 0; i < s.length(); i++) {
                char ch = s.charAt(i);
                if (ch != '#') {
                    ans1[size1++] = ch;
                } else {
                    if(size1!=0){
                        size1--;
                    }
                }
            }
            for (int j = 0; j < t.length(); j++) {
                char ch = t.charAt(j);
                if (ch != '#') {
                    ans2[size2++] = ch;
                } else {
                    if(size2!=0){
                        size2--;
                    }
                }
            }
            if (size1 == size2) {
                for (int i = 0; i < size1; i++) {
                    if (ans1[i] != ans2[i]) {
                        return false;
                    }
                }
            } else {
                return false;
            }
            return true;
        }
    }
```
